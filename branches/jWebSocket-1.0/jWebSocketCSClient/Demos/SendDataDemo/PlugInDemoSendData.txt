 private void sendData(WebSocketConnector aConnector, Token aToken) {
        
        Token lResponse = createResponse(aToken);

        if (aToken.getType().equals("list")) {
            List lData = aToken.getList("data");
            lResponse.setList("data", lData);

        } else if (aToken.getType().equals("dictionary")) {
            Map lData = aToken.getMap("data");
            lResponse.setMap("data", lData);

        } else if (aToken.getType().equals("string")) {
            String lData = aToken.getString("data");
            lResponse.setString("data", lData);

        } else if (aToken.getType().equals("double")) {
            double lData = aToken.getDouble("data");
            lResponse.setDouble("data", lData);

        } else if (aToken.getType().equals("int")) {
            int lData = aToken.getInteger("data");
            lResponse.setInteger("data", lData);

        } else if (aToken.getType().equals("bool")) {
            boolean lData = aToken.getBoolean("data");
            lResponse.setBoolean("data", lData);

        } else if (aToken.getType().equals("token")) {
            Token to = TokenFactory.createToken();
            to.setMap(aToken.getMap("data"));
            lResponse.setToken("data", to);
        } else {
            String lData = aToken.getString("data");
        }

        sendToken(aConnector, aConnector, lResponse);
    }
